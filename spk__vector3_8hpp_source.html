485  {

487  TType y_rad = p_rotationValues.y * static_cast<TType>(M_PI) / 180;
488  TType z_rad = p_rotationValues.z * static_cast<TType>(M_PI) / 180;
489 
490  TType cos_x = cos(x_rad);
491  TType sin_x = sin(x_rad);
492  TType cos_y = cos(y_rad);
493  TType sin_y = sin(y_rad);
494  TType cos_z = cos(z_rad);
495  TType sin_z = sin(z_rad);
496

498  result.x = cos_y * cos_z * x + (sin_x * sin_y * cos_z - cos_x * sin_z) * y + (cos_x * sin_y * cos_z + sin_x * sin_z) * z;
499  result.y = cos_y * sin_z * x + (sin_x * sin_y * sin_z + cos_x * cos_z) * y + (cos_x * sin_y * sin_z - sin_x * cos_z) * z;
